    <script>
        $(document).ready(function() {
        // Snowfall effect
            $(document).snowfall({
                image: "https://s6.imgcdn.dev/x7vn9.png",
                minSize: 15,
                maxSize: 15,
                maxSpeed: 1,
                flakeCount: 25
            });

            // Update counter
            function updateCounter() {
                const startDate = moment("2025-09-28");
                const currentDate = moment();
                
                // Calculate differences
                const years = currentDate.diff(startDate, "years");
                const months = currentDate.diff(startDate, "months") % 12;
                const days = currentDate.diff(startDate.clone().add(years, 'years').add(months, 'months'), "days");

                // Update DOM
                $("#yearResult").text(years);
                $("#monthResult").text(months);
                $("#dayResult").text(days);
            }

            // Initial update
            updateCounter();

            // Update every second
            setInterval(updateCounter, 1000);
        });
    </script>
